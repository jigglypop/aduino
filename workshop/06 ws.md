```python
class Circle:
    pi = 3.14
    x = 0
    y = 0
    r = 0
    
    def __init__(self, r, x, y):
        self.r = r
        self.x = x
        self.y = y        
    def area(self):
        return round(self.pi * self.r * self.r,2)
    def circumference(self):
        return round(2 * self.pi * self.r,2)
    def center(self):
        return (self.x, self.y)   
```


```python
c = Circle(5,0,0)
print(c.area())
print(c.circumference())
print(c.center())
```
